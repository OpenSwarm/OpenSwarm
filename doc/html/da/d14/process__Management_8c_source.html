390 void Sys_Add_Event_to_Process(uint16 pid, uint16 eventID, void *data, uint16 length){
391 
392  sys_pcb_list_element *element = Sys_Find_Process(pid);
393  if(element == 0){//no process with pid
394  return;
395  }
396

399  while(*o_event != 0){//check if the event (eventID) already occurred
400  if((*o_event)->eventID == eventID){//it already occurred

402  break;
403  }
404  o_event = &((*o_event)->next);
405  }
406 
407  if(add_event){//if it hasn't occurred

414  }

418  Sys_End_AtomicSection();
419  }
420 
421  //NOW add the data

423  while( (event = Sys_Next_EventHandler(event, eventID)) != 0 ){
424 
425  //check if the condition was met to add the event data

427  if(event->condition != 0){
428  is_condition_met = event->condition(data);

431  }
432

434  sys_event_data *e_data = (sys_event_data*) Sys_Malloc(sizeof(sys_event_data));
435  if(e_data == 0){//if malloc fails .. exit
436  return;
437  }
438

442  if(e_data->value == 0){//if malloc fails .. exit
443  Sys_Free(e_data);
444  return;
445  }
446 
447  Sys_Memcpy(data, e_data->value, length);
448  }else{
449  e_data->value = 0;
450  }
451  e_data->size = length;
452  e_data->next = 0;
453 
454  //add the struct to the end of the buffered_data
455  if(event->buffered_data == 0){
456  event->buffered_data = e_data;
457  }else{
458  sys_event_data *set_data = event->buffered_data;

460  set_data = set_data->next;
461  }

610  }

613  }
