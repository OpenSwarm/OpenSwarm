376 void Sys_Add_Event_to_Process(uint pid, uint eventID, void *data, uint length){
377 
378  sys_pcb_list_element *element = Sys_Find_Process(pid);
379  if(element == 0){//no process with pid
380  return;
381  }
382 
383  bool add_event = true;
384  sys_occurred_event **o_event = &sys_occurred_events;
385  while(*o_event != 0){//check if the event (eventID) already occurred
386  if((*o_event)->eventID == eventID){//it already occurred

391  }
392

400  }

407  //NOW add the data

410

412  bool is_condition_met = false;

414  is_condition_met = event->condition(data);
415  }else{//no condition is always met
416  is_condition_met = true;
417  }
418

421  if(e_data == 0){//if malloc fails .. exit
422  return;
423  }
424

426  if(length != 0){//if there is data

430  return;
431  }
432 
433  Sys_Memcpy(data, e_data->value, length);
434  }else{
435  e_data->value = 0;
436  }
437  e_data->size = length;
438  e_data->next = 0;
439 
440  //add the struct to the end of the buffered_data
441  if(event->buffered_data == 0){
442  event->buffered_data = e_data;
443  }else{
444  sys_event_data *set_data = event->buffered_data;

446  set_data = set_data->next;
